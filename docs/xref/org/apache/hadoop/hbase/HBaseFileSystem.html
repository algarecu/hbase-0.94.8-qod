33  /**
34   * An abstraction of the underlying filesystem. This is used by other entities such as
35   * {@link HLogFileSystem}, to make calls to the underlying filesystem.
36   *
37   */
38  public abstract class HBaseFileSystem {
39  
40    public static final Log LOG = LogFactory.getLog(HBaseFileSystem.class);
41  
42    /**
43     * In order to handle NN connectivity hiccups, one need to retry non-idempotent operation at the
44     * client level.
45     */
46    protected static int hdfsClientRetriesNumber;
47    private static int baseSleepBeforeRetries;
48    private static final int DEFAULT_HDFS_CLIENT_RETRIES_NUMBER = 10;
49    private static final int DEFAULT_BASE_SLEEP_BEFORE_RETRIES = 1000;
50    // This static block is added for performance reasons. This is to ensure we are not checking
51    // in the method calls whether retry properties are set or not. Refer to HBase-8288 for more 
52    // context.
53    static {
54      setRetryCounts(HBaseConfiguration.create());
55    }
56  
57    /**
58     * Deletes a file. Assumes the user has already checked for this directory existence.
59     * @param fs
60     * @param dir
61     * @return true if the directory is deleted.
62     * @throws IOException
63     */
64    public static boolean deleteFileFromFileSystem(FileSystem fs, Path dir)
65        throws IOException {
66      IOException lastIOE = null;
67      int i = 0;
68      do {
69        try {
70          return fs.delete(dir, false);
71        } catch (IOException ioe) {
72          lastIOE = ioe;
73          if (!fs.exists(dir)) return true;
74          // dir is there, retry deleting after some time.
75          sleepBeforeRetry("Delete File", i + 1);
76        }
77      } while (++i <= hdfsClientRetriesNumber);
78      throw new IOException("Exception in deleteFileFromFileSystem", lastIOE);
79    }

83     * Deletes a directory. Assumes the user has already checked for this directory existence.
84     * @param fs
85     * @param dir
86     * @return true if the directory is deleted.
87     * @throws IOException
88     */
89    public static boolean deleteDirFromFileSystem(FileSystem fs, Path dir)
90        throws IOException {
91      IOException lastIOE = null;
92      int i = 0;
93      do {
94        try {
95          return fs.delete(dir, true);
96        } catch (IOException ioe) {
97          lastIOE = ioe;
98          if (!fs.exists(dir)) return true;
99          // dir is there, retry deleting after some time.
100         sleepBeforeRetry("Delete Dir", i + 1);
101       }
102     } while (++i <= hdfsClientRetriesNumber);
103     throw new IOException("Exception in deleteDirFromFileSystem", lastIOE);
104   }

106   protected static void setRetryCounts(Configuration conf) {
107       hdfsClientRetriesNumber = conf.getInt("hdfs.client.retries.number",
108         DEFAULT_HDFS_CLIENT_RETRIES_NUMBER);
109       baseSleepBeforeRetries = conf.getInt("hdfs.client.sleep.before.retries",
110         DEFAULT_BASE_SLEEP_BEFORE_RETRIES);
111   }

113   /**
114    * Creates a directory for a filesystem and configuration object. Assumes the user has already
115    * checked for this directory existence.
116    * @param fs
117    * @param dir
118    * @return the result of fs.mkdirs(). In case underlying fs throws an IOException, it checks
119    *         whether the directory exists or not, and returns true if it exists.
120    * @throws IOException
121    */
122   public static boolean makeDirOnFileSystem(FileSystem fs, Path dir)
123       throws IOException {
124     int i = 0;
125     IOException lastIOE = null;
126     do {
127       try {
128         return fs.mkdirs(dir);
129       } catch (IOException ioe) {
130         lastIOE = ioe;
131         if (fs.exists(dir)) return true; // directory is present
132         sleepBeforeRetry("Create Directory", i+1);
133       }
134     } while (++i <= hdfsClientRetriesNumber);
135     throw new IOException("Exception in makeDirOnFileSystem", lastIOE);
136   }

138   /**
139    * Renames a directory. Assumes the user has already checked for this directory existence.
140    * @param fs
141    * @param src
142    * @param dst
143    * @return true if the directory is renamed.
144    * @throws IOException
145    */
146   public static boolean renameDirForFileSystem(FileSystem fs, Path src, Path dst)
147       throws IOException {
148     IOException lastIOE = null;
149     int i = 0;
150     do {
151       try {
152         return fs.rename(src, dst);
153       } catch (IOException ioe) {
154         lastIOE = ioe;
155         if (!fs.exists(src) && fs.exists(dst)) return true;
156         // src is there, retry renaming after some time.
157         sleepBeforeRetry("Rename Directory", i + 1);
158       }
159     } while (++i <= hdfsClientRetriesNumber);
160     throw new IOException("Exception in renameDirForFileSystem", lastIOE);
161   }

163   /**
164    * Creates a path on the file system. Checks whether the path exists already or not, and use it
165    * for retrying in case underlying fs throws an exception. 
166    * If the dir already exists and overwrite flag is false, the underlying FileSystem throws
167    *  an IOE. It is not retried and the IOE is re-thrown to the caller.
168    * @param fs
169    * @param dir
170    * @param overwrite
171    * @return
172    * @throws IOException
173    */
174   public static FSDataOutputStream createPathOnFileSystem(FileSystem fs, Path dir,
175       boolean overwrite) throws IOException {
176     int i = 0;
177     boolean existsBefore = fs.exists(dir);
178     IOException lastIOE = null;
179     do {
180       try {
181         return fs.create(dir, overwrite);
182       } catch (IOException ioe) {
183         lastIOE = ioe;
184         if (existsBefore && !overwrite) throw ioe;// a legitimate exception
185         sleepBeforeRetry("Create Path", i + 1);
186       }
187     } while (++i <= hdfsClientRetriesNumber);
188     throw new IOException("Exception in createPathOnFileSystem", lastIOE);
189   }
190 
191   /**
192    * Creates the specified file with the given permission. 
193    * If the dir already exists and the overwrite flag is false, underlying FileSystem throws
194    * an IOE. It is not retried and the IOE is re-thrown to the caller.
195    * @param fs
196    * @param path
197    * @param perm
198    * @param overwrite
199    * @return
200    * @throws IOException
201    */
202   public static FSDataOutputStream createPathWithPermsOnFileSystem(FileSystem fs,
203       Path path, FsPermission perm, boolean overwrite) throws IOException {
204     int i = 0;
205     IOException lastIOE = null;
206     boolean existsBefore = fs.exists(path);
207     do {
208       try {
209         return fs.create(path, perm, overwrite, FSUtils.getDefaultBufferSize(fs),
210           FSUtils.getDefaultReplication(fs, path), FSUtils.getDefaultBlockSize(fs, path), null);
211       } catch (IOException ioe) {
212         lastIOE = ioe;
213         if (existsBefore && !overwrite) throw ioe;// a legitimate exception
214         sleepBeforeRetry("Create Path with Perms", i + 1);
215       }
216     } while (++i <= hdfsClientRetriesNumber);
217     throw new IOException("Exception in createPathWithPermsOnFileSystem", lastIOE);
218   }

220 /**
221  * Creates the file. Assumes the user has already checked for this file existence.
222  * @param fs
223  * @param dir
224  * @return result true if the file is created with this call, false otherwise.
225  * @throws IOException
226  */
227   public static boolean createNewFileOnFileSystem(FileSystem fs, Path file)
228       throws IOException {
229     int i = 0;
230     IOException lastIOE = null;
231     do {
232       try {
233         return fs.createNewFile(file);
234       } catch (IOException ioe) {
235         lastIOE = ioe;
236         if (fs.exists(file)) return true; // file exists now, return true.
237         sleepBeforeRetry("Create NewFile", i + 1);
238       }
239     } while (++i <= hdfsClientRetriesNumber);
240     throw new IOException("Exception in createNewFileOnFileSystem", lastIOE);
241   }

243   /**
244    * sleeping logic for static methods; handles the interrupt exception. Keeping a static version
245    * for this to avoid re-looking for the integer values.
246    */
247   protected static void sleepBeforeRetry(String msg, int sleepMultiplier) {
248     if (sleepMultiplier > hdfsClientRetriesNumber) {
249       LOG.warn(msg + ", retries exhausted");
250       return;
251     }
252     LOG.info(msg + ", sleeping " + baseSleepBeforeRetries + " times " + sleepMultiplier);
253     Threads.sleep(baseSleepBeforeRetries * sleepMultiplier);
254   }
